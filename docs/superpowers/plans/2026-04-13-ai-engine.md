# AI Engine Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a new `nexus-ai` crate with AI provider traits, Anthropic/OpenAI/Ollama implementations, block-level content chunking, SQLite vector storage, RAG query pipeline, and CLI commands.

**Architecture:** Provider-agnostic traits (`AiProvider`, `EmbeddingProvider`) with HTTP implementations via reqwest. Embeddings stored as BLOBs in the existing SQLite `index.db` via schema migration v4. RAG pipeline embeds questions, searches for similar chunks via brute-force cosine similarity, builds context, and queries the AI provider. CLI replaces the existing `ai` stub.

**Tech Stack:** reqwest 0.12 (new), async-trait (existing), rusqlite (existing), tokio (existing), serde/serde_json (existing)

/// Errors from the AI engine.
#[derive(Debug, thiserror::Error)]
pub enum AiError {
    /// API key missing or invalid.
    #[error("authentication failed: {0}")]
    AuthFailed(String),

    /// HTTP request failed.
    #[error("network error: {0}")]
    Network(String),

    /// Provider returned an error response.
    #[error("provider error: {0}")]
    Provider(String),

    /// No provider configured.
    #[error("no AI provider configured — set ANTHROPIC_API_KEY, OPENAI_API_KEY, or OLLAMA_BASE_URL")]
    NoProvider,

    /// No embedding provider configured.
    #[error("no embedding provider configured — set OPENAI_API_KEY or OLLAMA_BASE_URL")]
    NoEmbeddingProvider,

    /// Database error.
    #[error("database error: {0}")]
    Database(#[from] rusqlite::Error),

    /// Serialization error.
    #[error("serialization error: {0}")]
    Serialization(String),
}

impl From<reqwest::Error> for AiError {
    fn from(e: reqwest::Error) -> Self {
        AiError::Network(e.to_string())

/// Configuration for an AI provider.
#[derive(Debug, Clone)]
pub struct AiConfig {
    /// Provider name: "anthropic", "openai", or "ollama".
    pub provider: String,
    /// Model override (uses provider default if None).
    pub model: Option<String>,
    /// API key (from env var or config).
    pub api_key: Option<String>,
    /// Base URL override (mainly for Ollama).
    pub base_url: Option<String>,
    /// Maximum tokens in the response.
    pub max_tokens: u32,
}

impl Default for AiConfig {
    fn default() -> Self {
        Self {
            provider: String::new(),
            model: None,
            api_key: None,
            base_url: None,
            max_tokens: 4096,
        }
    }
}

/// Detect the AI provider from environment variables.
///
/// Returns a config with the provider name and API key set,
/// or `None` if no provider is detected.
pub fn detect_provider() -> Option<AiConfig> {
    if let Ok(key) = std::env::var("ANTHROPIC_API_KEY") {
        return Some(AiConfig {
            provider: "anthropic".to_string(),
            api_key: Some(key),
            ..Default::default()
        });
    }
    if let Ok(key) = std::env::var("OPENAI_API_KEY") {
        return Some(AiConfig {
            provider: "openai".to_string(),
            api_key: Some(key),
            ..Default::default()
        });
    }
    if std::env::var("OLLAMA_BASE_URL").is_ok() || is_ollama_available() {
        return Some(AiConfig {
            provider: "ollama".to_string(),
            base_url: std::env::var("OLLAMA_BASE_URL").ok(),
            ..Default::default()
        });
    }
    None
}

//! Anthropic Claude provider implementation.

use async_trait::async_trait;
use reqwest::Client;
use serde::{Deserialize, Serialize};

use crate::{AiError, AiProvider, ChatMessage, Role};

/// Anthropic Claude provider.
pub struct AnthropicProvider {
    client: Client,
    api_key: String,
    model: String,
    max_tokens: u32,
}

impl AnthropicProvider {
    /// Create a new Anthropic provider.
    pub fn new(api_key: String, model: Option<String>, max_tokens: u32) -> Self {
        Self {
            client: Client::new(),
            api_key,
            model: model.unwrap_or_else(|| "claude-sonnet-4-20250514".to_string()),
            max_tokens,
        }
    }
}

#[derive(Serialize)]
struct AnthropicRequest {
    model: String,
    max_tokens: u32,
    #[serde(skip_serializing_if = "Option::is_none")]
    system: Option<String>,
    messages: Vec<AnthropicMessage>,
}

#[derive(Serialize)]
struct AnthropicMessage {
    role: String,
    content: String,
}

#[derive(Deserialize)]
struct AnthropicResponse {
    content: Vec<ContentBlock>,
}

#[derive(Deserialize)]
struct ContentBlock {
    text: Option<String>,
}

#[async_trait]
impl AiProvider for AnthropicProvider {
    async fn chat(
        &self,
        messages: &[ChatMessage],
        system: Option<&str>,
    ) -> Result<String, AiError> {
        let api_messages: Vec<AnthropicMessage> = messages
            .iter()
            .filter(|m| !matches!(m.role, Role::System))
            .map(|m| AnthropicMessage {
                role: match m.role {
                    Role::User => "user".to_string(),
                    Role::Assistant => "assistant".to_string(),
                    Role::System => unreachable!(),
                },
                content: m.content.clone(),
            })
            .collect();

        let request = AnthropicRequest {
            model: self.model.clone(),
            max_tokens: self.max_tokens,
            system: system.map(String::from),
            messages: api_messages,
        };

        let response = self
            .client
            .post("https://api.anthropic.com/v1/messages")
            .header("x-api-key", &self.api_key)
            .header("anthropic-version", "2023-06-01")
            .header("content-type", "application/json")
            .json(&request)
            .send()
            .await?;

        if !response.status().is_success() {
            let status = response.status();
            let body = response.text().await.unwrap_or_default();
            if status.as_u16() == 401 {
                return Err(AiError::AuthFailed("invalid Anthropic API key".to_string()));
            }
            return Err(AiError::Provider(format!("Anthropic API error {status}: {body}")));
        }

        let resp: AnthropicResponse = response.json().await?;
        let text = resp
            .content
            .into_iter()
            .filter_map(|b| b.text)
            .collect::<Vec<_>>()
            .join("");

        Ok(text)
    }

    fn model_name(&self) -> &str {
        &self.model
    }
}

#[derive(Serialize)]
struct OpenAiChatRequest {
    model: String,
    max_tokens: u32,
    messages: Vec<OpenAiMessage>,

#[derive(Serialize)]
struct OpenAiMessage {

    content: String,

#[derive(Deserialize)]
struct OpenAiResponseMessage {
    content: Option<String>,

#[async_trait]
impl AiProvider for OpenAiProvider {
    async fn chat(
        &self,
        messages: &[ChatMessage],
        system: Option<&str>,
    ) -> Result<String, AiError> {
        let mut api_messages = Vec::new();

        if let Some(sys) = system {
            api_messages.push(OpenAiMessage {
                role: "system".to_string(),
                content: sys.to_string(),
            });
        }

        for msg in messages {
            api_messages.push(OpenAiMessage {
                role: match msg.role {
                    Role::System => "system".to_string(),
                    Role::User => "user".to_string(),
                    Role::Assistant => "assistant".to_string(),
                },
                content: msg.content.clone(),
            });
        }

        let request = OpenAiChatRequest {
            model: self.chat_model.clone(),
            max_tokens: self.max_tokens,
            messages: api_messages,
        };

        let response = self
            .client
            .post("https://api.openai.com/v1/chat/completions")
            .header("Authorization", format!("Bearer {}", self.api_key))
            .json(&request)
            .send()
            .await?;

        if !response.status().is_success() {
            let status = response.status();
            let body = response.text().await.unwrap_or_default();
            if status.as_u16() == 401 {
                return Err(AiError::AuthFailed("invalid OpenAI API key".to_string()));
            }
            return Err(AiError::Provider(format!("OpenAI API error {status}: {body}")));
        }

        let resp: OpenAiChatResponse = response.json().await?;
        let text = resp
            .choices
            .first()
            .and_then(|c| c.message.content.clone())
            .unwrap_or_default();

        Ok(text)
    }

    fn model_name(&self) -> &str {
        &self.chat_model
    }
}

pub struct Chunk {

    pub block_id: u64,
    /// Content to embed (includes heading context).
    pub content: String,
}

/// Split parsed blocks into chunks suitable for embedding.
///
/// Each block becomes one chunk. Blocks exceeding `max_chunk_size` chars
/// are split on sentence boundaries. Each chunk includes the nearest
/// heading as context prefix.
pub fn chunks_from_blocks(
    file_path: &str,
    blocks: &[(u64, String, String, Option<i32>)], // (id, block_type, content, level)
    max_chunk_size: usize,
) -> Vec<Chunk> {
    let mut current_heading = String::new();
    let mut chunks = Vec::new();

    for (id, block_type, content, level) in blocks {
        if block_type == "heading" {
            if let Some(lvl) = level {
                let prefix = "#".repeat(*lvl as usize);
                current_heading = format!("{prefix} {content}");
            }
            continue; // Don't embed headings as standalone chunks
        }

        if content.trim().is_empty() {
            continue;
        }

        let contextualized = if current_heading.is_empty() {
            content.clone()
        } else {
            format!("{}\n\n{content}", current_heading)
        };

        if contextualized.len() <= max_chunk_size {
            chunks.push(Chunk {
                file_path: file_path.to_string(),
                block_id: *id,
                content: contextualized,
            });
        } else {
            // Split on sentence boundaries
            for sentence_chunk in split_on_sentences(&contextualized, max_chunk_size) {
                chunks.push(Chunk {
                    file_path: file_path.to_string(),
                    block_id: *id,
                    content: sentence_chunk,
                });
            }
        }
    }

    chunks
}

/// Split text into chunks at sentence boundaries.
fn split_on_sentences(text: &str, max_size: usize) -> Vec<String> {
    let mut chunks = Vec::new();
    let mut current = String::new();

    for sentence in text.split_inclusive('. ') {
        if current.len() + sentence.len() > max_size && !current.is_empty() {
            chunks.push(current.trim().to_string());
            current = String::new();
        }
        current.push_str(sentence);
    }

    if !current.trim().is_empty() {
        chunks.push(current.trim().to_string());
    }

    chunks
}

use rusqlite::{Connection, params};

use crate::AiError;

/// An embedding ready for storage.
#[derive(Debug, Clone)]
pub struct ChunkEmbedding {
    /// Path of the source file.
    pub file_path: String,
    /// Block ID from the storage index.
    pub block_id: u64,
    /// Text content that was embedded.
    pub chunk_text: String,
    /// Embedding vector.
    pub embedding: Vec<f32>,
}

/// A search result from vector similarity search.
#[derive(Debug, Clone)]
pub struct ChunkMatch {
    /// Path of the source file.
    pub file_path: String,
    /// Block ID from the storage index.
    pub block_id: u64,
    /// Text content of the matching chunk.
    pub chunk_text: String,
    /// Cosine similarity score (0.0 to 1.0).
    pub score: f32,
}

/// Insert or replace embeddings for a file.
///
/// # Errors
///
/// Returns [`AiError::Database`] on SQLite failure.
pub fn upsert(
    conn: &Connection,
    file_path: &str,
    chunks: &[ChunkEmbedding],
) -> Result<(), AiError> {
    // Delete existing embeddings for this file
    conn.execute(
        "DELETE FROM embeddings WHERE file_path = ?1;",
        params![file_path],
    )?;

    let now = now_unix();
    for chunk in chunks {
        let blob = embedding_to_blob(&chunk.embedding);
        conn.execute(
            "INSERT INTO embeddings (file_path, block_id, chunk_text, embedding, created_at)
             VALUES (?1, ?2, ?3, ?4, ?5);",
            params![
                chunk.file_path,
                chunk.block_id as i64,
                chunk.chunk_text,
                blob,
                now,
            ],
        )?;
    }

    Ok(())
}

/// Delete all embeddings for a file.
///
/// # Errors
///
/// Returns [`AiError::Database`] on SQLite failure.
pub fn delete_by_file(conn: &Connection, file_path: &str) -> Result<(), AiError> {
    conn.execute(
        "DELETE FROM embeddings WHERE file_path = ?1;",
        params![file_path],
    )?;
    Ok(())
}

/// Search for the most similar chunks to a query embedding.
///
/// Loads all embeddings and computes cosine similarity in Rust.
/// Returns top-K results sorted by descending similarity.
///
/// # Errors
///
/// Returns [`AiError::Database`] on SQLite failure.
pub fn search(
    conn: &Connection,
    query_embedding: &[f32],
    limit: usize,
) -> Result<Vec<ChunkMatch>, AiError> {
    let mut stmt = conn.prepare(
        "SELECT file_path, block_id, chunk_text, embedding FROM embeddings;"
    )?;

    let mut scored: Vec<ChunkMatch> = stmt
        .query_map([], |row| {
            let file_path: String = row.get(0)?;
            let block_id: i64 = row.get(1)?;
            let chunk_text: String = row.get(2)?;
            let blob: Vec<u8> = row.get(3)?;
            Ok((file_path, block_id as u64, chunk_text, blob))
        })?
        .filter_map(|r| r.ok())
        .map(|(file_path, block_id, chunk_text, blob)| {
            let embedding = blob_to_embedding(&blob);
            let score = cosine_similarity(query_embedding, &embedding);
            ChunkMatch {
                file_path,
                block_id,
                chunk_text,
                score,
            }
        })
        .collect();

    scored.sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap_or(std::cmp::Ordering::Equal));
    scored.truncate(limit);

    Ok(scored)
}

/// Count total embeddings in the store.
///
/// # Errors
///
/// Returns [`AiError::Database`] on SQLite failure.
pub fn count(conn: &Connection) -> Result<usize, AiError> {
    let n: i64 = conn.query_row(
        "SELECT COUNT(*) FROM embeddings;",
        [],
        |row| row.get(0),
    )?;
    Ok(n as usize)
}

// ── Helpers ──

/// Serialize f32 vector to little-endian bytes.
fn embedding_to_blob(embedding: &[f32]) -> Vec<u8> {
    embedding.iter().flat_map(|f| f.to_le_bytes()).collect()
}

/// Deserialize little-endian bytes to f32 vector.
fn blob_to_embedding(blob: &[u8]) -> Vec<f32> {
    blob.chunks_exact(4)
        .map(|b| f32::from_le_bytes([b[0], b[1], b[2], b[3]]))
        .collect()
}

/// Compute cosine similarity between two vectors.
fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
    let dot: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
    let norm_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
    let norm_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
    if norm_a == 0.0 || norm_b == 0.0 {
        return 0.0;
    }
    dot / (norm_a * norm_b)
}

fn now_unix() -> i64 {
    use std::time::{SystemTime, UNIX_EPOCH};
    SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs() as i64
}

//! RAG (Retrieval-Augmented Generation) pipeline.

use rusqlite::Connection;

use crate::{
    AiError, AiProvider, ChatMessage, EmbeddingProvider, Role,
    vectorstore::{self, ChunkEmbedding, ChunkMatch},
    chunker,
};

/// Response from a RAG query.
#[derive(Debug, Clone)]
pub struct RagResponse {
    /// The AI-generated answer.
    pub answer: String,
    /// Source chunks used as context.
    pub sources: Vec<ChunkMatch>,
    /// Model name that generated the answer.
    pub model: String,
}

/// Execute a RAG query: embed the question, find relevant chunks, ask the AI.
///
/// # Errors
///
/// Returns [`AiError`] on embedding, search, or provider failure.
pub async fn query(
    conn: &Connection,
    ai: &dyn AiProvider,
    embedder: &dyn EmbeddingProvider,
    question: &str,
    limit: usize,
) -> Result<RagResponse, AiError> {
    // 1. Embed the question
    let query_embeddings = embedder.embed(&[question.to_string()]).await?;
    let query_vec = query_embeddings
        .into_iter()
        .next()
        .ok_or_else(|| AiError::Provider("no embedding returned".to_string()))?;

    // 2. Search for similar chunks
    let sources = vectorstore::search(conn, &query_vec, limit)?;

    // 3. Build system prompt with context
    let system_prompt = build_rag_prompt(&sources);

    // 4. Ask the AI
    let messages = vec![ChatMessage {
        role: Role::User,
        content: question.to_string(),
    }];

    let answer = ai.chat(&messages, Some(&system_prompt)).await?;

    Ok(RagResponse {
        answer,
        sources,
        model: ai.model_name().to_string(),
    })
}

/// Index a single file: chunk its blocks, embed, and store.
///
/// `blocks` is a vec of `(block_id, block_type, content, level)` tuples.
///
/// # Errors
///
/// Returns [`AiError`] on embedding or database failure.
pub async fn index_file(
    conn: &Connection,
    embedder: &dyn EmbeddingProvider,
    file_path: &str,
    blocks: &[(u64, String, String, Option<i32>)],
) -> Result<usize, AiError> {
    let chunks = chunker::chunks_from_blocks(file_path, blocks, 2000);

    if chunks.is_empty() {
        return Ok(0);
    }

    // Embed all chunks
    let texts: Vec<String> = chunks.iter().map(|c| c.content.clone()).collect();
    let embeddings = embedder.embed(&texts).await?;

    // Build chunk embeddings
    let chunk_embeddings: Vec<ChunkEmbedding> = chunks
        .into_iter()
        .zip(embeddings)
        .map(|(chunk, embedding)| ChunkEmbedding {
            file_path: chunk.file_path,
            block_id: chunk.block_id,
            chunk_text: chunk.content,
            embedding,
        })
        .collect();

    let count = chunk_embeddings.len();
    vectorstore::upsert(conn, file_path, &chunk_embeddings)?;

    Ok(count)
}

/// Build the RAG system prompt from retrieved chunks.
fn build_rag_prompt(sources: &[ChunkMatch]) -> String {
    if sources.is_empty() {
        return "You are a helpful assistant. Answer the user's question based on your knowledge.".to_string();
    }

    let mut prompt = String::from(
        "You are a helpful assistant answering questions about a knowledge base. \
         Use the following context to answer the user's question. \
         Cite sources using [[filename]] notation when referencing specific content.\n\n\
         --- Context ---\n\n"
    );

    for (i, source) in sources.iter().enumerate() {
        prompt.push_str(&format!(
            "[{}] From [[{}]]:\n{}\n\n",
            i + 1,
            source.file_path,
            source.chunk_text
        ));
    }

    prompt.push_str("--- End Context ---\n\nAnswer based on the context above.");

    prompt
}

    Ok(())
}

/// Show current AI configuration.
pub fn config() -> Result<()> {
    let ai = nexus_ai::detect_provider();
    let embed = nexus_ai::detect_embedding_provider();

    println!("--- AI Provider ---");
    match ai {
        Some(c) => {
            println!("Provider : {}", c.provider);
            println!("Model    : {}", c.model.unwrap_or_else(|| "default".to_string()));
            println!("API Key  : {}", if c.api_key.is_some() { "set" } else { "not set" });
            if let Some(url) = c.base_url {
                println!("Base URL : {url}");
            }
        }
        None => println!("Not configured"),
    }

    println!("\n--- Embedding Provider ---");
    match embed {
        Some(c) => {
            println!("Provider : {}", c.provider);
            println!("API Key  : {}", if c.api_key.is_some() { "set" } else { "not set" });
        }
        None => println!("Not configured"),
    }

    Ok(())
}
